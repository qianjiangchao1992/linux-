package javaBasics;
import java.util.Scanner
public class ScannerDemo{
	private int age;
	public ScannerDemo(String designation){
		System.out.println("请输入岗位: "+designation);
		}
	public void SetAge(int age){
		if(age>0 && age<120)
		{
		this.age=age;
		}
		else
		{
		System.out.println("输入年龄不符合要求");
		}}
	public int GetAge(){
		return age;}
	public static void main(String []args){
		Scanner scan= Scanner new(System.in);
		Scanner scan1=Scanner new(System.in);
		try{
		int ScannerDemo.age= scan.next();
		String designation=scan1.next();
		ScannerDemo(designation);
		ScannerDemo.SetAge(ScannerDemo.age);
		System.out.println("年龄: "+ScannerDemo.GetAge());}
		catch(Exception e){
            System.out.println("出现异常："+e);    //打印异常信息
            e.printStackTrace(); 
		}finally{
            System.out.println("不管是否出现异常都执行此代码");
        }
        System.out.println("计算结束");}}
